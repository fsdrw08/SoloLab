done by cli command
```powershell
# get atlantis user password
$credential = Get-Credential -Message "credential to login vault" -UserName "000"
$env:VAULT_ADDR = "https://vault.day1.sololab"
vault login -no-print -method=ldap username=$($credential.UserName) password=$($credential.GetNetworkCredential().Password)
$atlantisUserName="bot-atlantis"
$atlantisPassword=$(vault kv get -format=json -mount=kvv2_gitea user-$atlantisUserName | jq.exe .data.data.password).Replace('"', '')
$tokenName="atlantis-token"

# prepare request body
$requestBody = @"
{
  "name": "$tokenName",
  "scopes": [
    "write:issue",
    "write:repository",
    "read:user"
  ]
}
"@ | jq -c

$giteaURI = "https://gitea.day4.sololab/api/v1/users/$atlantisUserName/tokens"

# check token
$tokenStatusCode = curl.exe -k -s -L -I `
    --request GET `
    -u "$($atlantisUserName):$($atlantisPassword)" `
    -H "Content-Type: application/json" `
    --output NUL `
    -w "%{http_code}\n" `
    $giteaURI

# delete token if exists
if ($tokenStatusCode -eq "200") {
    $token = $(curl.exe -k -s `
        --request DELETE `
        -u "$($atlantisUserName):$($atlantisPassword)" `
        -H "Content-Type: application/json" `
        $giteaURI/$tokenName)
}

# create token
# https://github.com/cloudbase/garm/blob/47811d0046a9515085b85d88038f87ac68fd793c/doc/gitea.md?plain=1#L127
$token = $(curl.exe -k -s `
    --request POST `
    -u "$($atlantisUserName):$($atlantisPassword)" `
    -H "Content-Type: application/json" `
    -d "$($requestBody)" `
    $giteaURI | jq.exe -r '.sha1').Replace('"', '')

# store token in vault
vault kv patch -mount=kvv2_gitea user-$atlantisUserName token="$token"
vault kv patch -mount=kvv2_others app-atlantis gitea_token="$token"
```
